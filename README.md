# lambda-calculus

## input

    f = "a"
    g = 42
    id = (\x -> x)
    j = (id g)

## output

    ====================================
    "f" = "a"
    "g" = 42
    "id" = (\"x" -> "x")
    "j" = ("id" "g")
    ====================================
    f0 = "a"
    g1 = 42
    id2 = (\x4 -> x4)
    j3 = (id2 g1)
    ====================================
